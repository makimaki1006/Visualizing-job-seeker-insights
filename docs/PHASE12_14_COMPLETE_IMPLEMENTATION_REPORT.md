def _generate_supply_demand_gap(self):
    """éœ€çµ¦ã‚®ãƒ£ãƒƒãƒ—åˆ†æãƒ‡ãƒ¼ã‚¿ã‚’ç”Ÿæˆï¼ˆMAPçµ±åˆå¯¾å¿œï¼‰"""

    gap = gap[[
